<!-- Padrão de controller (apps/api/src/modules/auth/auth.controller.ts): -->
export class AuthController {
  private service: AuthService
  constructor(private fastify: FastifyInstance) {
    this.service = new AuthService(fastify)
  }
  async register(request: FastifyRequest, reply: FastifyReply) {
    let body: ReturnType<typeof RegisterSchema.parse>
    try {
      body = RegisterSchema.parse(request.body)
    } catch (err) {
      if (err instanceof ZodError) return reply.status(400).send({ error: zodMessage(err) })
      return reply.status(400).send({ error: 'Dados inválidos.' })
    }
    try {
      const result = await this.service.register(body)
      if ('error' in result) return reply.status(result.status).send({ error: result.error })
      return reply.status(201).send(result)
    } catch (err) {
      this.fastify.log.error(err)
      return reply.status(500).send({ error: 'Erro interno. Tente novamente.' })
    }
  }
}

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| cliente → POST /payments/pix | Entrada não confiável — amount calculado no servidor (nunca aceito do cliente) |
| cliente → POST /payments/card | Token vem do Brick MP (seguro); demais campos validados por Zod |
| cliente → PUT /users/me/auto-recharge | Body validado por AutoRechargeSchema; salva apenas preferência, sem pagamento |
| cliente → PUT /users/me/card-token | Body validado por CardTokenSchema; token é gerado pelo Brick MP no navegador |
| API → Mercado Pago | Chave secreta MP_ACCESS_TOKEN no servidor, nunca exposta ao frontend |
