<!-- Validação de webhook MP com HMAC-SHA256 (RESEARCH.md Pattern 3): -->
import { createHmac } from 'node:crypto'

function validateMPSignature(xSignature: string, xRequestId: string, dataId: string, secret: string): boolean {
  const parts = Object.fromEntries(xSignature.split(',').map(p => p.split('=')))
  const { ts, v1 } = parts
  const manifest = `id:${dataId};request-id:${xRequestId};ts:${ts};`
  const computed = createHmac('sha256', secret).update(manifest).digest('hex')
  return computed === v1
}

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Mercado Pago → POST /webhooks/mercadopago | Endpoint público — qualquer HTTP request pode chegar aqui |
| webhooks.service → PaymentsRepository | Crédito ao saldo é operação crítica — única que modifica creditBalance |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-03-01 | Spoofing | POST /webhooks/mercadopago | mitigate | HMAC-SHA256 com MP_WEBHOOK_SECRET validado ANTES de qualquer lógica; retorna 401 se inválida (ASVS V4) |
| T-03-02 | Tampering | WebhooksService.processApprovedPayment | mitigate | Idempotência: checar Payment.status === 'PAID' antes de creditar; $transaction atômico para creditBalance + CreditTransaction |
| T-03-03 | Elevation of Privilege | creditUserBalance | mitigate | Quantidade creditada lida do Combo.quantity ou Payment.customQuantity no banco — nunca do payload do webhook |
| T-03-06 | Repudiation | WebhooksService | accept | Logs de webhook com fastify.log.error em falhas — auditoria básica suficiente para MVP |
</threat_model>
